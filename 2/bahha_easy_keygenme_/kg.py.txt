import sys

def gks(username):
	keybuf = bytearray(username)
	sz = len(username)
	for i in xrange(sz):
		for j in xrange(sz):
			if keybuf[i] == 'z':
				keybuf[i] = 'a'
			elif keybuf[i] == 'Z':
				keybuf[i] = 'A'
			else:
				keybuf[i] += 1
	return keybuf

def generate(username):
	print gks(username)

if __name__ == '__main__':
	if len(sys.argv) != 2:
		print 'Usage: python kg.py <username>'
	else:
		generate(sys.argv[1])
